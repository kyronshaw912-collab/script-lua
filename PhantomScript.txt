-- NOCLIP FUNCTIONS
local function StopNoclip()
    Noclip.active = false
    if Noclip.conn then Noclip.conn:Disconnect() Noclip.conn = nil end
end

local function StartNoclip()
    Noclip.active = true
    Noclip.conn = RunService.Stepped:Connect(function()
        local c = getChar()
        if not c then return end
        for _, p in ipairs(c:GetDescendants()) do
            if p:IsA("BasePart") then p.CanCollide = false end
        end
    end)
end

local function ToggleNoclip()
    if Noclip.active then StopNoclip() else StartNoclip() end
end

-- FLY FUNCTIONS
local function StopFly()
    Fly.active = false
    if Fly.conn then Fly.conn:Disconnect() Fly.conn = nil end
    if Fly.bv then Fly.bv:Destroy() Fly.bv = nil end
    if Fly.bg then Fly.bg:Destroy() Fly.bg = nil end
    local h = getHum() if h then h.PlatformStand = false end
    Fly.vel = Vector3.new(0, 0, 0)
end

local function StartFly()
    local r = getHRP() local h = getHum()
    if not r or not h then return end
    Fly.active = true
    h.PlatformStand = true
    Fly.bv = Instance.new("BodyVelocity")
    Fly.bv.Velocity = Vector3.new(0, 0, 0)
    Fly.bv.MaxForce = Vector3.new(1e5, 1e5, 1e5)
    Fly.bv.P = 1e4
    Fly.bv.Parent = r
    Fly.bg = Instance.new("BodyGyro")
    Fly.bg.MaxTorque = Vector3.new(1e5, 1e5, 1e5)
    Fly.bg.P = 1e4
    Fly.bg.Parent = r
    Fly.conn = RunService.Heartbeat:Connect(function()
        local r2 = getHRP() if not r2 then return end
        local spd = Config.FlySpeed * (Fly.sprinting and Config.SprintMult or 1)
        local cf = Cam.CFrame
        local dir = Vector3.new(0, 0, 0)
        if UIS:IsKeyDown(Enum.KeyCode.W) then dir = dir + cf.LookVector end
        if UIS:IsKeyDown(Enum.KeyCode.S) then dir = dir - cf.LookVector end
        if UIS:IsKeyDown(Enum.KeyCode.A) then dir = dir - cf.RightVector end
        if UIS:IsKeyDown(Enum.KeyCode.D) then dir = dir + cf.RightVector end
        if UIS:IsKeyDown(Config.UpKey) then dir = dir + Vector3.new(0, 1, 0) end
        if UIS:IsKeyDown(Config.DownKey) then dir = dir - Vector3.new(0, 1, 0) end
        local target = dir.Magnitude > 0 and dir.Unit * spd or Vector3.new(0, 0, 0)
        Fly.vel = Fly.vel:Lerp(target, Config.Smoothness)
        Fly.bv.Velocity = Fly.vel
        Fly.bg.CFrame = cf
    end)
end

local function ToggleFly()
    if Fly.active then StopFly() else StartFly() end
end

local function MakeTracerLine(color)
    local f = Instance.new("Frame", TracerGui)
    f.BackgroundColor3 = color or Color3.fromRGB(255, 60, 60)
    f.BorderSizePixel = 0
    f.AnchorPoint = Vector2.new(0, 0.5)
    f.Visible = false
    f.ZIndex = 5
    return f
end

local function SetTracerLine(line, from, to, thickness, color)
    local diff = to - from
    local length = diff.Magnitude
    if length < 1 then line.Visible = false return end
    line.Size = UDim2.new(0, length, 0, thickness or 2)
    line.Position = UDim2.new(0, from.X, 0, from.Y)
    line.Rotation = math.deg(math.atan2(diff.Y, diff.X))
    if color then line.BackgroundColor3 = color end
    line.Visible = true
end

local function GetDistanceColor(dist)
    if not Config.DistanceColors then return nil end
    if dist < 100 then
        return Config.NearColor
    elseif dist < 300 then
        return Config.MidColor
    else
        return Config.FarColor
    end
end

local function BuildChams(player)
    local obj = ESP.objects[player]
    if not obj then return end
    for _, sb in ipairs(obj.chams) do pcall(function() sb:Destroy() end) end
    obj.chams = {}
    local c = getChar(player)
    if not c then return end
    for _, part in ipairs(c:GetDescendants()) do
        if part:IsA("BasePart") and part.Name ~= "HumanoidRootPart" then
            local sb = Instance.new("SelectionBox")
            sb.Adornee = part
            sb.Color3 = Config.ChamsColor
            sb.LineThickness = 0.05
            sb.SurfaceTransparency = Config.ChamsFillAlpha
            sb.SurfaceColor3 = Config.ChamsFill
            sb.Parent = ESPFolder
            table.insert(obj.chams, sb)
        end
    end
end

local function SetChams(obj, on)
    for _, sb in ipairs(obj.chams or {}) do
        sb.SurfaceTransparency = on and Config.ChamsFillAlpha or 1
        sb.LineThickness = on and 0.05 or 0
    end
end

local function RemoveESP(player)
    local obj = ESP.objects[player]
    if not obj then return end
    if obj.billboard then pcall(function() obj.billboard:Destroy() end) end
    if obj.box then pcall(function() obj.box:Destroy() end) end
    if obj.headDot then pcall(function() obj.headDot:Destroy() end) end
    if obj.tracerLine then pcall(function() obj.tracerLine:Destroy() end) end
    for _, b in ipairs(obj.beams) do
        pcall(function() b.attA:Destroy() end)
        pcall(function() b.attB:Destroy() end)
        pcall(function() b.beam:Destroy() end)

    for _, d in ipairs(obj.drawings2d or {}) do

    end

end

local rainbowHue = 0
local function UpdateESP()
    local myPos = (getHRP() and getHRP().Position) or Vector3.new(0, 0, 0)
    local vp = Cam.ViewportSize
    local tracerOrigin
    if Config.TracerFrom == "Bottom" then
        tracerOrigin = Vector2.new(vp.X / 2, vp.Y)
    elseif Config.TracerFrom == "Center" then
        tracerOrigin = Vector2.new(vp.X / 2, vp.Y / 2)
    else
        tracerOrigin = UIS:GetMouseLocation()
    end
    
    if Config.RainbowESP then
        local speeds = {0.3, 0.8, 2.0}
        rainbowHue = (rainbowHue + (speeds[Config.RainbowSpeed] or 0.3)) % 360
    end
    
    local rc = nil
    if Config.RainbowESP then
        local t = rainbowHue / 360
        local m = Config.RainbowMode
        if m == 1 then rc = Color3.fromHSV(t, 1, 1)
        elseif m == 2 then rc = Color3.new(1, 0.1 + 0.4 * (math.sin(t * math.pi * 2) * 0.5 + 0.5), 0.1)
        elseif m == 3 then rc = Color3.new(0.1, 0.1 + 0.4 * (math.sin(t * math.pi * 2) * 0.5 + 0.5), 1)
        elseif m == 4 then rc = Color3.new(0.1, 1, 0.1 + 0.4 * (math.sin(t * math.pi * 2) * 0.5 + 0.5))
        else rc = Color3.fromHSV(t, 1, 1) end
    end
    
    for player, obj in pairs(ESP.objects) do
        local c = getChar(player)
        local h = c and c:FindFirstChildOfClass("Humanoid")
        local hrp = c and c:FindFirstChild("HumanoidRootPart")
        local alive = c and h and hrp and h.Health > 0
        local dist = alive and (hrp.Position - myPos).Magnitude or math.huge
        local show = alive and dist <= Config.MaxDist
        
        -- Get appropriate color
        local teamCol = Config.TeamColor and player.Team and player.Team.TeamColor.Color
        local distCol = GetDistanceColor(dist)
        local finalColor = rc or teamCol or distCol or Config.BoxColor
        
        if obj.billboard then
            obj.billboard.Enabled = show
            if show then
                if obj.nameBG then
                    obj.nameBG.Visible = Config.ShowName
                    obj.nameBG.BackgroundColor3 = Color3.fromRGB(0, 0, 0)
                end
                if obj.nameStroke then
                    obj.nameStroke.Color = finalColor
                end
                if obj.nameLbl then
                    obj.nameLbl.Text = player.DisplayName
                    obj.nameLbl.TextColor3 = finalColor
                end
                if obj.hpTrack then
                    obj.hpTrack.Visible = Config.ShowHealth
                end
                if obj.hpFill and h then
                    local ratio = math.clamp(h.Health / math.max(h.MaxHealth, 1), 0, 1)
                    obj.hpFill.Size = UDim2.new(ratio, 0, 1, 0)
                    obj.hpFill.BackgroundColor3 = Color3.new(
                        math.clamp(2 * (1 - ratio), 0, 1),
                        math.clamp(2 * ratio, 0, 1),
                        0
                    )
                end
                if obj.hpText and h then
                    obj.hpText.Visible = Config.ShowHealthText
                    obj.hpText.Text = string.format("%d HP", math.floor(h.Health))
                    obj.hpText.TextColor3 = finalColor
                end
                if obj.distLbl then
                    obj.distLbl.Visible = Config.ShowDistance
                    obj.distLbl.Text = string.format("[%d studs]", math.floor(dist))
                    obj.distLbl.TextColor3 = finalColor
                end
            end
        end
        
        if obj.box then
            obj.box.Visible = show and Config.ShowBox and Config.BoxType == "3D"
            if show and Config.ShowBox then
                obj.box.Color3 = finalColor
                obj.box.SurfaceColor3 = finalColor
            end
        end
        
        if obj.headDot then
            obj.headDot.Visible = show and Config.ShowHeadDot
            if show and Config.ShowHeadDot then
                obj.headDot.Color3 = finalColor
            end
        end
        
        local showSkel = show and Config.ShowSkeleton
        for _, b in ipairs(obj.beams) do
            if b.beam and b.beam.Parent then
                b.beam.Enabled = showSkel
                if showSkel then
                    b.beam.Color = ColorSequence.new(finalColor)
                end
            end
        end
        
        if obj.tracerLine then
            if show and Config.ShowTracer and hrp then
                local sp, on = w2s(hrp.Position)
                if on then
                    SetTracerLine(obj.tracerLine, tracerOrigin, sp, 2, finalColor)
                else
                    obj.tracerLine.Visible = false
                end
            else
                obj.tracerLine.Visible = false
            end
        end
        
        if show and Config.ShowChams then
            if #obj.chams == 0 then BuildChams(player) end
            SetChams(obj, true)
            for _, sb in ipairs(obj.chams) do
                sb.Color3 = finalColor
                sb.SurfaceColor3 = Color3.fromHSV(select(1, finalColor:ToHSV()), 0.6, 0.4)
            end
        else
            SetChams(obj, false)
        end
    end
end

local function EnableESP()
    ESP.active = true
    for _, p in ipairs(Players:GetPlayers()) do
        MakeESP(p)
        BuildESPOnChar(p)
        BuildChams(p)
    end
    ESP.conn = RunService.Heartbeat:Connect(UpdateESP)
end

local function DisableESP()
    ESP.active = false
    if ESP.conn then ESP.conn:Disconnect() ESP.conn = nil end
    for p in pairs(ESP.objects) do RemoveESP(p) end
end

local function ToggleESP()
    if ESP.active then DisableESP() else EnableESP() end
end

local function IsVisible(targetPos)
    local myC = getChar()
    local origin = Cam.CFrame.Position
    local dir = targetPos - origin
    local params = RaycastParams.new()
    params.FilterDescendantsInstances = myC and {myC} or {}
    params.FilterType = Enum.RaycastFilterType.Exclude
    local result = workspace:Raycast(origin, dir.Unit * dir.Magnitude, params)
    if not result then return true end
    if (result.Position - targetPos).Magnitude < 2 then return true end
    return false
end

local function GetAimbotTarget()
    local center = Cam.ViewportSize / 2
    local bestDist = Config.AimbotFOV
    local bestPart = nil
    for _, p in ipairs(Players:GetPlayers()) do
        if p ~= LP then
            if not (Config.AimbotTeamCheck and p.Team == LP.Team) then
                local c = getChar(p)
                local h = c and c:FindFirstChildOfClass("Humanoid")
                if c and h and h.Health > 0 then
                    local part = c:FindFirstChild(Config.AimbotBone) or c:FindFirstChild("Head") or c:FindFirstChild("HumanoidRootPart")
                    if part then
                        local sp, on = w2s(part.Position)
                        if on then
                            local sd = (sp - center).Magnitude
                            if sd <= Config.AimbotFOV then
                                if not Config.AimbotWallCheck or IsVisible(part.Position) then
                                    if sd < bestDist then bestDist = sd bestPart = part end
                                end
                            end
                        end
                    end
                end
            end
        end
    end
    return bestPart
end

local function StartAimbot()
    Aim.active = true
    RunService:BindToRenderStep("NSAimbot", Enum.RenderPriority.Last.Value, function()
        if not Aim.active then return end
        if UIS:IsMouseButtonPressed(Enum.UserInputType.MouseButton2) then return end
        local target = GetAimbotTarget()
        if not target then return end
        local targetCF = CFrame.new(Cam.CFrame.Position, target.Position)
        Cam.CFrame = Cam.CFrame:Lerp(targetCF, Config.AimbotSmooth)
    end)
end

local function StopAimbot()
    Aim.active = false
    pcall(function() RunService:UnbindFromRenderStep("NSAimbot") end)
    if Aim.conn then Aim.conn:Disconnect() Aim.conn = nil end
end

local function ToggleAimbot()
    if Aim.active then StopAimbot() else StartAimbot() end
end
