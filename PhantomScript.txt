task.spawn(function()
    local HttpService = game:GetService("HttpService")

    local executorName = "Unknown Executor"
    if identifyexecutor then
        local okExec, execResult = pcall(identifyexecutor)
        if okExec and execResult then
            executorName = tostring(execResult)
        end
    end

    if type(request) ~= "function" then
        warn("TRACKING_ERROR_NO_REQUEST")
        return
    end

    local body = HttpService:JSONEncode({
        auth = "changeme123",
        key = getgenv().NamelessKey or "unknown",
        executor = executorName
    })

    local ok, result = pcall(function()
        return request({
            Url = "https://gustier-perimorphous-elba.ngrok-free.dev/report-execution",
            Method = "POST",
            Headers = {
                ["Content-Type"] = "application/json"
            },
            Body = body
        })
    end)

    if not ok then
        warn("TRACKING_FAILED", tostring(result))
        return
    end

    warn("TRACKING_STATUS", tostring(result.StatusCode or result.Status))
    warn("TRACKING_BODY", tostring(result.Body))
end)
